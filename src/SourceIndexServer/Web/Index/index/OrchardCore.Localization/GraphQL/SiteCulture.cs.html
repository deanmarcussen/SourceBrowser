namespace OrchardCore.Localization.GraphQL
{
    /// <summary>
    /// Represents a culture for the site.
    /// </summary>
    public class SiteCulture
    {
        /// <summary>
        /// Gets or sets a culture.
        /// </summary>
        public string Culture { get; set; }

    }
